016package nl.knaw.dans.lib.dataverse;
017
018import com.fasterxml.jackson.databind.ObjectMapper;
019import nl.knaw.dans.lib.dataverse.model.ResumeMessage;
020import org.apache.http.HttpResponse;
021import org.apache.http.client.HttpClient;
022import org.apache.http.client.methods.HttpPost;
023import org.apache.http.entity.StringEntity;
024
025import java.io.IOException;
026import java.net.URI;
027
028public class WorkflowsApi {
029
030    private static final ObjectMapper mapper = new ObjectMapper();
031
032    private final URI baseUrl;
033    private final HttpClient httpClient;
034
035    public WorkflowsApi(URI baseUrl, HttpClient httpClient) {
036        this.baseUrl = baseUrl;
037        this.httpClient = httpClient;
038    }
039
040    public HttpResponse resume(String invocationId, ResumeMessage resumeMessage) throws IOException {
041        HttpPost post = new HttpPost(baseUrl.resolve(invocationId));
042        post.setEntity(new StringEntity(mapper.writeValueAsString(resumeMessage)));
043        return httpClient.execute(post);
044    }
045
046}
